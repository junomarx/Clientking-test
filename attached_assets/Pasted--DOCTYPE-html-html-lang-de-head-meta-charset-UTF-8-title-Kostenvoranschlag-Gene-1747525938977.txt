    <script>
        function generateEstimate() {
            const form = document.getElementById('estimateForm');
            if (!form.checkValidity()) {
                alert('Bitte füllen Sie alle Pflichtfelder aus.');
                return;
            }

            const data = {
                vorname: document.getElementById('vorname').value,
                nachname: document.getElementById('nachname').value,
                adresse: document.getElementById('adresse').value,
                plz: document.getElementById('plz').value,
                ort: document.getElementById('ort').value,
                telefon: document.getElementById('telefon').value,
                email: document.getElementById('email').value,
                geraetetyp: document.getElementById('geraetetyp').value,
                hersteller: document.getElementById('hersteller').value,
                modell: document.getElementById('modell').value,
                seriennummer: document.getElementById('seriennummer').value,
                fehlerbeschreibung: document.getElementById('fehlerbeschreibung').value,
                arbeiten: document.getElementById('arbeiten').value,
                gesamtpreis: document.getElementById('gesamtpreis').value
            };

            alert('Kostenvoranschlag wurde erstellt für: ' + data.vorname + ' ' + data.nachname +
                  '\\nAdresse: ' + data.adresse + ', ' + data.plz + ' ' + data.ort +
                  '\\nTelefon: ' + data.telefon + ', E-Mail: ' + data.email +
                  '\\nGerät: ' + data.hersteller + ' ' + data.modell + ' (' + data.geraetetyp + ')' +
                  '\\nGesamtpreis: ' + data.gesamtpreis);
        }
    </script>
